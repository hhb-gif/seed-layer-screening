# src/seed_layer/config.py 末尾添加
def save_config(config_dict: dict, path: str) -> None:
    """Save config dictionary to YAML file.

    Args:
        config_dict: Configuration dictionary to save
        path: Output YAML file path
    """
    import yaml
    path = Path(path)
    path.parent.mkdir(parents=True, exist_ok=True)
    with open(path, "w", encoding="utf-8") as f:
        yaml.dump(config_dict, f, default_flow_style=False, allow_unicode=True)
```

with tab_run:
    st.subheader("运行控制")

    col1, col2 = st.columns(2)
    with col1:
        # 材料列表选择
        data_dir = Path("data")
        material_files = ["(默认: MP API 查询)"]
        if data_dir.exists():
            material_files += [f.name for f in data_dir.glob("*.txt")]
        selected_file = st.selectbox("材料列表", material_files)
        materials_path = None
        if selected_file != "(默认: MP API 查询)":
            materials_path = str(data_dir / selected_file)

    with col2:
        run_tag = st.text_input("运行标签 (可选)", placeholder="zn-test")
        skip_neb = st.checkbox("跳过 NEB 扩散计算")

    # ── 进程管理 ──
    process = st.session_state.get("pipeline_process")
    run_dir = st.session_state.get("pipeline_run_dir")

    col_start, col_stop = st.columns(2)
    with col_start:
        if st.button("▶ 开始筛选", use_container_width=True, disabled=process is not None):
            # 先保存配置
            save_config(config, CONFIG_PATH)

            # 构建命令
            cmd = [sys.executable, "src/main.py", "--config", CONFIG_PATH]
            if run_tag:
                cmd += ["--tag", run_tag]
            if materials_path:
                cmd += ["--materials", materials_path]
            if skip_neb:
                cmd += ["--skip-neb"]

            # 启动后台进程
            import subprocess
            proc = subprocess.Popen(
                cmd,
                stdout=subprocess.PIPE,
                stderr=subprocess.STDOUT,
                cwd=str(Path(__file__).parent),
            )
            st.session_state.pipeline_process = proc
            st.session_state.pipeline_start_time = time.time()
            st.rerun()

    with col_stop:
        if st.button("⏹ 停止", use_container_width=True, disabled=process is None):
            if process:
                process.terminate()
                st.session_state.pipeline_process = None
                st.warning("已停止")
                st.rerun()

    # ── 状态显示 ──
    if process is not None:
        import time

        returncode = process.poll()
        elapsed = time.time() - st.session_state.get("pipeline_start_time", time.time())
        elapsed_str = f"{int(elapsed // 60)}m {int(elapsed % 60)}s"

        if returncode is None:
            st.success(f"🟢 运行中 | 已用时: {elapsed_str}")
        else:
            if returncode == 0:
                st.success(f"✅ 完成 | 总用时: {elapsed_str}")
            else:
                st.error(f"❌ 失败 (exit code {returncode}) | 用时: {elapsed_str}")
            st.session_state.pipeline_process = None

        # 显示日志
        st.subheader("实时日志")
        log_placeholder = st.empty()

        # 找到最新的 output 目录
        output_dir = Path("output")
        if output_dir.exists():
            runs = sorted(output_dir.iterdir(), key=lambda p: p.stat().st_mtime, reverse=True)
            if runs:
                log_file = runs[0] / "logs" / "run.log"
                if log_file.exists():
                    with open(log_file, "r", encoding="utf-8") as f:
                        lines = f.readlines()[-30:]  # 最后 30 行
                    log_placeholder.code("".join(lines))
                else:
                    log_placeholder.info("日志文件尚未生成...")
    else:
        st.info("点击「开始筛选」启动 pipeline")
```

                import matplotlib
                matplotlib.use("Agg")
                import matplotlib.pyplot as plt

                chart_cols = st.columns(2)

                # 吸附能分布
                with chart_cols[0]:
                    if "E_ads_eV" in df.columns and df["E_ads_eV"].notna().any():
                        fig, ax = plt.subplots(figsize=(6, 4))
                        df["E_ads_eV"].dropna().hist(ax=ax, bins=20, color="steelblue", edgecolor="white")
                        ax.axvline(x=-0.5, color="red", linestyle="--", label="最优 -0.5 eV")
                        ax.set_xlabel("吸附能 (eV)")
                        ax.set_ylabel("材料数")
                        ax.set_title("吸附能分布")
                        ax.legend()
                        st.pyplot(fig)
                        plt.close()

                # 扩散势垒分布
                with chart_cols[1]:
                    if "barrier_eV" in df.columns and df["barrier_eV"].notna().any():
                        fig, ax = plt.subplots(figsize=(6, 4))
                        df["barrier_eV"].dropna().hist(ax=ax, bins=20, color="coral", edgecolor="white")
                        ax.set_xlabel("扩散势垒 (eV)")
                        ax.set_ylabel("材料数")
                        ax.set_title("扩散势垒分布")
                        st.pyplot(fig)
                        plt.close()

                # 界面能 vs 吸附能散点图
                if "interface_energy_eV_per_A2" in df.columns and "E_ads_eV" in df.columns:
                    scatter_df = df.dropna(subset=["interface_energy_eV_per_A2", "E_ads_eV"])
                    if not scatter_df.empty:
                        fig, ax = plt.subplots(figsize=(6, 4))
                        ax.scatter(
                            scatter_df["E_ads_eV"],
                            scatter_df["interface_energy_eV_per_A2"],
                            c=scatter_df.get("score", "steelblue"),
                            cmap="RdYlGn", s=60, edgecolors="gray",
                        )
                        ax.set_xlabel("吸附能 (eV)")
                        ax.set_ylabel("界面能 (eV/Å²)")
                        ax.set_title("界面能 vs 吸附能")
                        st.pyplot(fig)
                        plt.close()

                # 评分雷达图（Top 1 材料）
                if "score" in df.columns and len(df) > 0:
                    top1 = df.iloc[0]
                    radar_labels = ["吸附", "扩散", "界面", "晶格", "稳定性"]
                    radar_keys = ["S_adsorption", "S_diffusion", "S_interface", "S_lattice", "S_stability"]
                    radar_values = [top1.get(k, 0) or 0 for k in radar_keys]

                    import numpy as np
                    angles = np.linspace(0, 2 * np.pi, len(radar_labels), endpoint=False).tolist()
                    radar_values += radar_values[:1]
                    angles += angles[:1]

                    fig, ax = plt.subplots(figsize=(5, 5), subplot_kw=dict(polar=True))
                    ax.fill(angles, radar_values, color="steelblue", alpha=0.25)
                    ax.plot(angles, radar_values, color="steelblue", linewidth=2)
                    ax.set_xticks(angles[:-1])
                    ax.set_xticklabels(radar_labels)
                    ax.set_ylim(0, 1)
                    ax.set_title(f"Top 1: {top1.get('formula', top1.get('material_id', '?'))}", y=1.08)
                    st.pyplot(fig)
                    plt.close()

            else:
                st.warning("该运行未生成 summary.csv")

            # 输出目录链接
            st.markdown(f"📂 输出目录: `{run_path.resolve()}`")
